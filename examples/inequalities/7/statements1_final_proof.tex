\begin{theorem}
Let $a, b, c > 0$ such that $a + b + c = abc$. Then
\[
\frac{1}{1+a^2} + \frac{1}{1+b^2} + \frac{1}{1+c^2} \le \frac{3}{2}.
\]
\end{theorem}

\begin{claim}
Under the same conditions,
\[
\frac{1}{\sqrt{1+a^{2}}}+\frac{1}{\sqrt{1+b^{2}}}+\frac{1}{\sqrt{1+c^{2}}}\le\frac{3}{2},
\]
with equality if and only if $a=b=c=\sqrt{3}$.
\end{claim}
\begin{proof}
Set $A = \arctan a$, $B = \arctan b$, $C = \arctan c$. Since $a,b,c > 0$, we have $A,B,C \in (0, \frac{\pi}{2})$. The condition $a+b+c = abc$ becomes
\[
\tan A + \tan B + \tan C = \tan A \tan B \tan C,
\]
which is equivalent to $A+B+C = \pi$. Moreover,
\[
\frac{1}{\sqrt{1+a^{2}}} = \cos A,\quad
\frac{1}{\sqrt{1+b^{2}}} = \cos B,\quad
\frac{1}{\sqrt{1+c^{2}}} = \cos C.
\]
The function $\cos x$ is concave on $[0, \frac{\pi}{2}]$ because its second derivative $-\cos x \le 0$ there. By Jensen's inequality for concave functions,
\[
\frac{\cos A + \cos B + \cos C}{3} \le \cos\!\left(\frac{A+B+C}{3}\right) = \cos\frac{\pi}{3} = \frac12.
\]
Hence $\cos A + \cos B + \cos C \le \frac32$, i.e.
\[
\frac{1}{\sqrt{1+a^{2}}}+\frac{1}{\sqrt{1+b^{2}}}+\frac{1}{\sqrt{1+c^{2}}}\le\frac{3}{2}.
\]
Equality in Jensen holds when $A = B = C$. Together with $A+B+C = \pi$ this gives $A = B = C = \frac{\pi}{3}$, so $a = b = c = \tan\frac{\pi}{3} = \sqrt{3}$. Conversely, if $a = b = c = \sqrt{3}$, the leftâ€‘hand side equals $\frac{3}{\sqrt{1+3}} = \frac{3}{2}$.
\end{proof}

\begin{claim}
For any $x > 0$, we have $\frac{1}{1+x^2} \le \frac{1}{\sqrt{1+x^2}}$.
\end{claim}
\begin{proof}
Since $1+x^2 > 0$, we have
\[
\frac{1}{1+x^2} \le \frac{1}{\sqrt{1+x^2}} \Longleftrightarrow \sqrt{1+x^2} \le 1+x^2 \Longleftrightarrow 1+x^2 \le (1+x^2)^2.
\]
But $(1+x^2)^2 = 1 + 2x^2 + x^4 \ge 1 + x^2$, with equality only when $x=0$. Since $x>0$, the inequality is strict.
\end{proof}

\begin{claim}
The desired inequality follows.
\end{claim}
\begin{proof}
Using Claim 2 for each term, we obtain
\[
\frac{1}{1+a^2} + \frac{1}{1+b^2} + \frac{1}{1+c^2}
\le \frac{1}{\sqrt{1+a^{2}}} + \frac{1}{\sqrt{1+b^{2}}} + \frac{1}{\sqrt{1+c^{2}}}.
\]
By Claim 1, the right-hand side is at most $\frac{3}{2}$. Therefore,
\[
\frac{1}{1+a^2} + \frac{1}{1+b^2} + \frac{1}{1+c^2} \le \frac{3}{2}.
\]
Equality in the original inequality would require equality in both Claim 1 and Claim 2. Equality in Claim 1 occurs only when $a=b=c=\sqrt{3}$, but for $x=\sqrt{3}>0$, Claim 2 gives a strict inequality. Hence the original inequality is strict for all $a,b,c>0$ satisfying $a+b+c=abc$.
\end{proof}
