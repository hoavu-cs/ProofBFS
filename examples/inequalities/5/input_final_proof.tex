\begin{theorem}
For positive real numbers $a$, $b$, $c$ satisfying $(a+b-c)\left(\frac{1}{a}+\frac{1}{b}-\frac{1}{c}\right)=3$, we have
\[
\bigl(a^{4}+b^{4}-c^{4}\bigr)\Bigl(\frac{1}{a^{4}}+\frac{1}{b^{4}}-\frac{1}{c^{4}}\Bigr)\ge 361+208\sqrt{3},
\]
with equality if and only if $ab=c^{2}$ and $\dfrac{a+b}{c}=1+\sqrt{3}$.
\end{theorem}

\begin{claim}
Set $x=\frac{a}{c}$, $y=\frac{b}{c}$. Then the condition becomes $(x+y-1)\left(\frac{1}{x}+\frac{1}{y}-1\right)=3$. Let $s=x+y$ and $t=xy$ (so $s>0$, $t>0$). Then
\[
t=\frac{s(s-1)}{s+2},\qquad s>1.
\]
\end{claim}
\begin{proof}
From $(x+y-1)\left(\frac{1}{x}+\frac{1}{y}-1\right)=3$, we have
\[
(s-1)\left(\frac{x+y}{xy}-1\right)=3 \implies (s-1)\left(\frac{s}{t}-1\right)=3.
\]
Solving for $t$:
\[
\frac{s}{t}-1 = \frac{3}{s-1} \implies \frac{s}{t} = 1 + \frac{3}{s-1} = \frac{s+2}{s-1} \implies t = \frac{s(s-1)}{s+2}.
\]
Since $x,y>0$, we have $s>0$, $t>0$. Moreover, from the original condition, $x+y-1>0$ and $\frac{1}{x}+\frac{1}{y}-1>0$, so $s>1$.
\end{proof}

\begin{claim}
Define $F(s) = \bigl(x^{4}+y^{4}-1\bigr)\Bigl(\frac{1}{x^{4}}+\frac{1}{y^{4}}-1\Bigr)$. Then
\[
F(s)=\frac{8s^{10}+8s^{9}+40s^{8}+97s^{7}+161s^{6}+251s^{5}+383s^{4}+456s^{3}+348s^{2}+160s+32}
      {s^{2}\bigl(s^{5}+s^{4}-5s^{3}-s^{2}+8s-4\bigr)},\qquad s>1.
\]
\end{claim}
\begin{proof}
We compute:
\[
x^{4}+y^{4} = (x^{2}+y^{2})^{2}-2x^{2}y^{2} = \bigl((x+y)^{2}-2xy\bigr)^{2}-2(xy)^{2} = (s^{2}-2t)^{2}-2t^{2},
\]
and
\[
\frac{1}{x^{4}}+\frac{1}{y^{4}} = \frac{x^{4}+y^{4}}{(xy)^{4}} = \frac{x^{4}+y^{4}}{t^{4}}.
\]
Thus
\[
F(s) = \bigl((s^{2}-2t)^{2}-2t^{2}-1\bigr)\left(\frac{(s^{2}-2t)^{2}-2t^{2}}{t^{4}}-1\right).
\]
Substituting $t = \frac{s(s-1)}{s+2}$ and simplifying (e.g., via algebraic manipulation) yields the stated rational expression.
\end{proof}

\begin{claim}
The derivative $F'(s)$ is given by
\[
F'(s)=\frac{8(3s^{12}+6s^{11}-8s^{10}-4s^{9}-68s^{8}-184s^{7}-336s^{6}-512s^{5}-580s^{4}-408s^{3}-128s^{2}+16s+16)}
          {s^{3}(s^{7}+2s^{6}-6s^{5}-8s^{4}+17s^{3}+6s^{2}-20s+8)}.
\]
The numerator factors as $(s^{2}+2)(s^{2}-2s-2)(s^{4}+2s^{2}+4s+2)(3s^{4}+12s^{3}+10s^{2}+4s-2)$, so the only real root of $F'(s)=0$ in $s>1$ is $s=1+\sqrt{3}$ (the positive root of $s^{2}-2s-2=0$).
\end{claim}
\begin{proof}
Differentiating $F(s)$ (or the factored form) yields the expression. The factorization shows that the critical points come from the factors. For $s>1$, the factors $s^{2}+2$, $s^{4}+2s^{2}+4s+2$, and $3s^{4}+12s^{3}+10s^{2}+4s-2$ are positive (the last one is positive for $s>1$). Thus the only critical point in $s>1$ is from $s^{2}-2s-2=0$, i.e., $s=1+\sqrt{3}$.
\end{proof}

\begin{claim}
The function $F(s)$ attains its global minimum on $(1,\infty)$ at $s=1+\sqrt{3}$, with value $F(1+\sqrt{3})=361+208\sqrt{3}$.
\end{claim}
\begin{proof}
We check the limits: $\lim\limits_{s\to1^{+}}F(s)=+\infty$ and $\lim\limits_{s\to\infty}F(s)=+\infty$. Since $F(s)$ is differentiable on $(1,\infty)$ and has only one critical point at $s=1+\sqrt{3}$, this point must be the global minimum. Evaluating $F(s)$ at $s=1+\sqrt{3}$ gives $361+208\sqrt{3}$.
\end{proof}

\begin{claim}
Equality in the theorem holds if and only if $ab=c^{2}$ and $\frac{a+b}{c}=1+\sqrt{3}$.
\end{claim}
\begin{proof}
From the substitution, $s=\frac{a+b}{c}$ and $t=\frac{ab}{c^{2}}$. At the minimum, $s=1+\sqrt{3}$ and $t=\frac{s(s-1)}{s+2}=1$. Thus $ab=c^{2}$ and $\frac{a+b}{c}=1+\sqrt{3}$. Conversely, if these conditions hold, then $s=1+\sqrt{3}$ and $t=1$, so $F(s)=361+208\sqrt{3}$, giving equality.
\end{proof}
